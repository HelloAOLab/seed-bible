const dimension = os.getCurrentDimension();
const mainWord = getBot("isMainWord", true);
let currentBibleStructure = mainWord.masks.bibleStructure.slice();
const radius = 4;
let currentPosition;

if(thisBot.tags.transformer)
{
    const transformerBot = getBot(byID(thisBot.tags.transformer));

    const transformerBotPosition = new Vector3(
        (transformerBot.masks[dimension + "X"] ?? transformerBot.tags[dimension + "X"]),
        (transformerBot.masks[dimension + "Y"] ?? transformerBot.tags[dimension + "Y"]),
        (transformerBot.masks[dimension + "Z"] ?? transformerBot.tags[dimension + "Z"])
    )
    currentPosition = new Vector3(
        thisBot.masks[dimension + "X"] ?? thisBot.tags[dimension + "X"],
        thisBot.masks[dimension + "Y"] ?? thisBot.tags[dimension + "Y"],
        thisBot.masks[dimension + "Z"] ?? thisBot.tags[dimension + "Z"]
    )
    currentPosition = new Vector3(
        currentPosition.x + transformerBotPosition.x,
        currentPosition.y + transformerBotPosition.y,
        currentPosition.z + transformerBotPosition.z + 1
    )
}
else
{
    currentPosition = new Vector3(
        thisBot.masks[dimension + "X"] ?? thisBot.tags[dimension + "X"],
        thisBot.masks[dimension + "Y"] ?? thisBot.tags[dimension + "Y"],
        thisBot.masks[dimension + "Z"] ?? thisBot.tags[dimension + "Z"]
    )
}
const mainWordPosition = new Vector3(
    mainWord.masks[dimension + "X"] ?? mainWord.tags[dimension + "X"],
    mainWord.masks[dimension + "Y"] ?? mainWord.tags[dimension + "Y"],
    mainWord.masks[dimension + "Z"] ?? mainWord.tags[dimension + "Z"]
)
let parentTestamentPosition = null;

for(let testament of currentBibleStructure)
{
    if(testament.convertedIntoArrangement)
    {
        if(testament.sections.some((section) => {return section.botID === getID(thisBot)}) && !testament.isInsideBible && testament.positionOutsideBible)
        {
            parentTestamentPosition = testament.positionOutsideBible;
        }
    }
}
const distanceBetweenBotAndBible = Vector3.distanceBetween(mainWordPosition, currentPosition);

if(parentTestamentPosition)
{
    const distanceBetweenBotAndTestament = Vector3.distanceBetween(parentTestamentPosition, currentPosition);
    if(distanceBetweenBotAndTestament <= radius || distanceBetweenBotAndBible <= radius)
    {
        if(distanceBetweenBotAndTestament < distanceBetweenBotAndBible)
        {
            if(distanceBetweenBotAndTestament <= radius)
            {
                // Closer to Testament
                if(thisBot.tryToIncludeInTestament())
                {
                    mainWord.resizeBible();
                }
            }
        }
        else
        {
            if(distanceBetweenBotAndBible <= radius)
            {
                // Closer to Bible
                if(thisBot.tryToIncludeInTheBible())
                {
                    mainWord.resizeBible();
                }
            }
        }
    }
}
else
{
    if(distanceBetweenBotAndBible <= radius)
    {
        // Closer to Bible
        if(thisBot.tryToIncludeInTheBible())
        {
            mainWord.resizeBible();
        }
    }
}
