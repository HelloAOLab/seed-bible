setTagMask(thisBot, "isInTransition", true);

const animationDuration = 0.5
const parentSection = getBot(byTag("isSection", true), byTag("sectionName", thisBot.tags.sectionName))

animateTag(thisBot, "formOpacity", null),
animateTag(thisBot, "labelOpacity", null),
animateTag(thisBot, "scaleX", null),
animateTag(thisBot, "scaleY", null),
animateTag(thisBot, "scaleZ", null)

await Promise.allSettled([
    animateTag(thisBot, "formOpacity", {
        toValue: thisBot.tags.unhoveredOpacity,
        duration: animationDuration,
        easing: {type: "sinusoidal", mode: "inout"}
    }),
    animateTag(thisBot, "labelOpacity", {
        toValue: 0,
        duration: animationDuration / 5,
        easing: {type: "sinusoidal", mode: "inout"}
    }),
    animateTag(thisBot, "scaleX", {
        toValue: thisBot.tags.isBookAsSection ? (thisBot.tags.initialScaleX * (thisBot.masks.hasBeenScaledAsBook ? 0.9 : 1)) : (parentSection.masks.isInExplodedView && thisBot.tags.explodedViewCustomScale ? (thisBot.tags.explodedViewCustomScale.x * parentSection.tags.initialScaleX) : thisBot.tags.initialScaleX),
        duration: animationDuration,
        easing: {type: "elastic", mode: "out"}
    }),
    animateTag(thisBot, "scaleY", {
        toValue: thisBot.tags.isBookAsSection ? (thisBot.tags.initialScaleY * (thisBot.masks.hasBeenScaledAsBook ? 0.9 : 1)) : (parentSection.masks.isInExplodedView && thisBot.tags.explodedViewCustomScale ? (thisBot.tags.explodedViewCustomScale.y * parentSection.tags.initialScaleY) : thisBot.tags.initialScaleY),
        duration: animationDuration,
        easing: {type: "elastic", mode: "out"}
    }),
    animateTag(thisBot, "scaleZ", {
        toValue: thisBot.tags.initialScaleZ,
        duration: animationDuration,
        easing: {type: "elastic", mode: "out"}
    })
])

setTagMask(thisBot, "isInTransition", false);
setTagMask(thisBot, "isExpanded", false);
setTagMask(thisBot, "isConvertedIntoATray", false);
