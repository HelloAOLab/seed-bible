function CreateExpandShadow(scaleX, scaleY, scaleZ, color)
{
    const expandShadow = create({
        space: "tempLocal",
        pointable: false,
        isBookExpandShadow: true,
        parentBookId: getID(thisBot),
        [dimension]: true,
        [dimension + "X"]: thisBot.masks[dimension + "X"] ?? thisBot.tags[dimension + "X"],
        [dimension + "Y"]: thisBot.masks[dimension + "Y"] ?? thisBot.tags[dimension + "Y"],
        [dimension + "Z"]: thisBot.masks[dimension + "Z"] ?? thisBot.tags[dimension + "Z"],
        color: color,
        scaleX: scaleX,
        scaleY: scaleY,
        scaleZ: scaleZ,
        formOpacity: 0.05,
        onCreate: `@
        setTagMask(thisBot, "isBlinking", true);
        const animationDuration = 0.3;

        while(thisBot.masks.isBlinking)
        {
            await animateTag(thisBot, "formOpacity", {
                toValue: 0.3,
                duration: animationDuration,
                easing: {type: "sinusoidal", mode: "inout"}
            })
            await animateTag(thisBot, "formOpacity", {
                toValue: 0.05,
                duration: animationDuration,
                easing: {type: "sinusoidal", mode: "inout"}
            })
        }
        `,
        onDestroy: `@setTagMask(thisBot, "isBlinking", false)`
    });
    return expandShadow;
}
