const dimension = os.getCurrentDimension();
const camRotation = os.getCameraRotation("grid");
const camForwardDirection = math.getForwardDirection(camRotation);
const camBackwardDirection = camForwardDirection.negate();
const ejectVectorMagnitude = 8;
const directionToEject = new Vector3(camBackwardDirection.x, camBackwardDirection.y, 0).normalize();
const mainWord = getBot("isMainWord", true);
let currentBibleStructure = mainWord.masks.bibleStructure.slice();
const animationDuration = 0.5;
const transformerBot = getBot(byID(thisBot.tags.transformer));
const currentPosition = new Vector3(
    thisBot.masks[dimension + "X"] ?? thisBot.tags[dimension + "X"],
    thisBot.masks[dimension + "Y"] ?? thisBot.tags[dimension + "Y"],
    thisBot.masks[dimension + "Z"] ?? thisBot.tags[dimension + "Z"],
);
const newPosition = new Vector3(
    currentPosition.x + (transformerBot.masks[dimension + "X"] ?? transformerBot.tags[dimension + "X"]),
    currentPosition.y + (transformerBot.masks[dimension + "Y"] ?? transformerBot.tags[dimension + "Y"]),
    currentPosition.z + (transformerBot.masks[dimension + "Z"] ?? transformerBot.tags[dimension + "Z"])
)
setTag(thisBot, "transformer", null);
setTagMask(thisBot, dimension + "X", newPosition.x);
setTagMask(thisBot, dimension + "Y", newPosition.y);
setTagMask(thisBot, dimension + "Z", newPosition.z);

thisBot.forceUnhighlight();
setTagMask(mainWord, "isInBibleAnimation", true);
setTagMask(thisBot, "highlightable", false);
setTagMask(thisBot, "isInsideBible", false);
setTagMask(thisBot, "isInsideTestament", false);
setTagMask(thisBot, "isInsideSection", false);
SoundsManager.playEjectBookSound();

for(let testament of currentBibleStructure)
{
    for(let section of testament.sections)
    {
        if(thisBot.tags.isBookAsSection)
        {
            if(section.botID === getID(thisBot))
            {
                section.isInsideBible = false;
                section.isInsideArrangement = false;
                setTagMask(mainWord, "bibleStructure", currentBibleStructure);
                mainWord.resizeBible();
            }
        }
        else
        {
            for(let element of section.books)
            {
                if(Array.isArray(element))
                {
                    for(let book of element)
                    {
                        if(book.botID === getID(thisBot))
                        {
                            book.isInsideBible = false;
                            book.isInsideArrangement = false;
                            book.isInsideSection = false;
                            setTagMask(mainWord, "bibleStructure", currentBibleStructure);
                            mainWord.resizeBible();
                        }
                    }
                }
                else
                {
                    if(element.botID === getID(thisBot))
                    {
                        element.isInsideBible = false;
                        element.isInsideArrangement = false;
                        element.isInsideSection = false;
                        setTagMask(mainWord, "bibleStructure", currentBibleStructure);
                        mainWord.resizeBible();
                    }
                }
            }
        }
    }
}

await Promise.allSettled([
    animateTag(thisBot, dimension + "X", {
        toValue: newPosition .x + (directionToEject.x * ejectVectorMagnitude),
        duration: animationDuration,
        easing: {type: "linear"}
    }),
    animateTag(thisBot, dimension + "Y", {
        toValue: newPosition.y + (directionToEject.y * ejectVectorMagnitude),
        duration: animationDuration,
        easing: {type: "linear"}
    }),
    animateTag(thisBot, dimension + "Z", {
        toValue: directionToEject.z,
        duration: animationDuration,
        easing: {type: "cubic", mode: "in"}
    }),
]);

setTagMask(mainWord, "isInBibleAnimation", false);
setTagMask(thisBot, "highlightable", true);
setTagMask(thisBot, "isOnTheGround", true);
setTagMask(thisBot, "canBeDraggedBackToParent", true);
setTag(thisBot, "desiredPositionZ", directionToEject.z);
