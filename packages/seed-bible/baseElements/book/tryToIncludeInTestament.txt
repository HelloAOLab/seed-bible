const mainWord = getBot("isMainWord", true);
if(!thisBot.masks.isInsideTestament && thisBot.masks.canBeDraggedBackToParent && !mainWord.masks.isBibleClosed)
{
    thisBot.tryToSetRegularShape();
    SoundsManager.playIncludeElementInParentSound();
    let currentBibleStructure = mainWord.masks.bibleStructure.slice();
    if(thisBot.tags.transformer)
    {
        const dimension = os.getCurrentDimension();
        const transformerBot = getBot(byID(thisBot.tags.transformer));
        const currentPosition = new Vector3(
            thisBot.masks[dimension + "X"] ?? thisBot.tags[dimension + "X"],
            thisBot.masks[dimension + "Y"] ?? thisBot.tags[dimension + "Y"],
            thisBot.masks[dimension + "Z"] ?? thisBot.tags[dimension + "Z"]
        )
        const transformerBotPosition = new Vector3(
            (transformerBot.masks[dimension + "X"] ?? transformerBot.tags[dimension + "X"]),
            (transformerBot.masks[dimension + "Y"] ?? transformerBot.tags[dimension + "Y"]),
            (transformerBot.masks[dimension + "Z"] ?? transformerBot.tags[dimension + "Z"])
        )
        const newPosition = new Vector3(
            currentPosition.x + transformerBotPosition.x,
            currentPosition.y + transformerBotPosition.y,
            currentPosition.z + transformerBotPosition.z + 1
        )
        setTag(thisBot, "transformer", null);
        setTagMask(thisBot, dimension + "X", newPosition.x);
        setTagMask(thisBot, dimension + "Y", newPosition.y);
        setTagMask(thisBot, dimension + "Z", newPosition.z);
        setTag(thisBot, "desiredPositionZ", newPosition.z);
    }
    setTagMask(thisBot, "isBeingIncludedInParent", true);
    setTagMask(thisBot, "isInsideTestament", true);
    setTagMask(thisBot, "canBeDraggedBackToParent", false);

    if(thisBot.tags.isBookAsSection)
    {
        for(let testament of currentBibleStructure)
        {
            for(let section of testament.sections)
            {
                if(section.botID === getID(thisBot))
                {
                    section.isInsideArrangement = true;
                    section.isConvertedIntoATray = false;
                }
            }
        }
    }
    else
    {
        for(let testament of currentBibleStructure)
        {
            for(let section of testament.sections)
            {
                for(let element of section.books)
                {
                    if(Array.isArray(element))
                    {
                        for(let book of element)
                        {
                            if(book.botID === getID(thisBot))
                            {
                                setTagMask(thisBot, "isInsideSection", section.isInsideBible);
                                book.isInsideSection = section.isInsideBible;
                                book.isInsideArrangement = true;
                                book.isConvertedIntoATray = false;
                            }
                        }
                    }
                    else
                    {
                        if(element.botID === getID(thisBot))
                        {
                            setTagMask(thisBot, "isInsideSection", section.isInsideBible);
                            element.isInsideSection = section.isInsideBible;
                            element.isInsideArrangement = true;
                            element.isConvertedIntoATray = false;
                        }
                    }
                }
            }
        }
    }

    if(thisBot.tags.isConvertedIntoATray)
    {
        let trayBookOutline = getBot(byTag("isTrayBookOutline", true), byTag("trayBookName", thisBot.tags.bookName));
        if(trayBookOutline)
        {
            ObjectPooler.ReleaseObject({obj: trayBookOutline, tag: trayBookOutline.tags.poolTag});
        }
        setTagMask(thisBot, "isConvertedIntoATray", false);
        // thisBot.lerpColor({startingColor: HexToRgb(thisBot.masks.color ?? thisBot.tags.color), endingColor: HexToRgb(thisBot.tags.initialColor), durationInSeconds: 0.75})
        LerpColorManager.LerpTagColor({startingColor: HexToRgb(thisBot.masks.color ?? thisBot.tags.color), endingColor: HexToRgb(thisBot.tags.initialColor), durationInSeconds: 0.75, bot: thisBot, tag: InterpolatableColorTags.Color});

    }
    
    setTagMask(mainWord, "bibleStructure", currentBibleStructure);

    if(thisBot.masks.isExpanded)
    {
        thisBot.displayShrinkAnimation();
    }

    return true
}
else
{
    return false;
}
