import {StackData} from 'managers.StacksManager.StackData'
import {TestamentData} from "managers.StacksManager.TestamentData"
import {SectionData} from "managers.StacksManager.SectionData"
import {BookData} from "managers.StacksManager.BookData"

const {element, data, parentData} = that;
const {testamentData, sectionData, bookData} = StacksManager.GetBibleElementData({bibleElement: element});
const dimension = os.getCurrentDimension();
setTagMask(thisBot, "isBibleAnimating", true);
setTagMask(element, "isBeingPushedToParent", true);
setTagMask(element, 'isBeingDragged', false)
setTagMask(element, "isDraggableToParent", false);
let parent;
switch (true)
{
    case parentData instanceof StackData:
        parent = parentData.staticBibleElements.bibleTransformer;
        data.isInsideBible = true;
        if(data instanceof SectionData || data instanceof BookData)
        {
            data.isInsideTestament = testamentData.isInsideBible;
            if(data instanceof BookData) data.isInsideSection = sectionData.isInsideBible;
        }
        break;
    case parentData instanceof TestamentData:
        parent = parentData.testament;
        data.isInsideBible = parentData.isInsideBible;
        data.isInsideTestament = true;
        if(data instanceof BookData) data.isInsideSection = sectionData.isInsideBible;
        break;
    case parentData instanceof SectionData:
        parent = parentData.section;
        data.isInsideSection = true;
        data.isInsideTestament = parentData.isInsideTestament;
        data.isInsideBible = parentData.isInsideBible;
        break;
    default: break;
}
if(!element.tags.transformer)
{
    const elementPosition = getBotPosition(element, dimension);
    const parentPosition = getBotPosition(parent, dimension);
    setTag(element, "transformer", getID(parent));
    const newPosition = elementPosition.subtract(parentPosition);
    setTagMask(element, dimension + "X", newPosition.x);
    setTagMask(element, dimension + "Y", newPosition.y);
    setTagMask(element, dimension + "Z", newPosition.z);
}
data.positionOutsideBible = null;

thisBot.UpdateStacks().then(() => {setTagMask(thisBot, "isBibleAnimating", false)});
