let arrangementsArray = thisBot.tags.arrangementsInfo.slice();
const arrangementIndex = 0;
const testamentIndex = 1;
const arrangement = arrangementsArray[arrangementIndex];
const testament = arrangement[testamentIndex];
const sectionKey = "Law";
const section = testament.sections[sectionKey];
for(let book of section)
{
    if(book.chaptersData) continue;
    const bookIndex = section.indexOf(book);
    let chaptersData = [];
    for(let i = 1; i < (book.numberOfChapters + 1); i++)
    {
        const chapterContent = await web.get(`https://bible.helloao.org/api/BSB/${book.commonName.replaceAll(" ", "_")}/${i}.json`).then((response) => {return response.data.chapter.content});
        const chapterVerses = chapterContent.filter((content) => {return content.type === "verse"});
        const amountOfVerses = chapterVerses.length;
        chaptersData.push(amountOfVerses);
    }
    chaptersData = chaptersData.map((amountOfVerses, index) => {return {number: (index+1), amountOfVerses: amountOfVerses}})
    console.log(chaptersData);
    arrangementsArray[arrangementIndex][testamentIndex].sections[sectionKey][bookIndex].chaptersData = chaptersData;
    setTag(thisBot, "arrangementsInfo", arrangementsArray);
}
