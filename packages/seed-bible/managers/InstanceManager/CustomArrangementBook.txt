const {useEffect, useState} = os.appHooks;

const DragOverBar = thisBot.DragOverBar();

const CustomArrangementBook = ({
    book,
    handleDragStart,
    handleDragEnter,
    handleDragEnd,
    index,
    handleItemDelete,
    handleItemColorChange,
    dragOverInfo,
    draggingId
}) => {

    return ( 
        <>
            { (dragOverInfo?.id === book.id && dragOverInfo.position === "top") && <DragOverBar />}

            <li 
                key={book.id} 
                draggable="true" 
                className={`listItem ${draggingId === book.id ? "dragging" : ""}`}
                onDragStart={() => {handleDragStart(book.id)}} 
                onDragEnter={() => {handleDragEnter(book.id)}} 
                onDragEnd={handleDragEnd}
                index={index}
                style={{
                    backgroundColor: book.color,
                    color: GetTextColorBasedOnBackground(book.color)
                }}
            >
                {book.name}
                <div className="actionButtonsContainer">
                    <span class={`material-symbols-outlined`} onClick={() => {handleItemDelete(index)}}>close</span>
                    <div>
                        <span class="material-symbols-outlined" >palette</span>
                        <input type="color" value={book.color} onChange={(e) => {handleItemColorChange(e, book.id)}}/>
                    </div>
                </div>
            </li> 
            {(dragOverInfo?.id === book.id && dragOverInfo.position === "bottom") && <DragOverBar />}
        </>
    )
}

return CustomArrangementBook;
