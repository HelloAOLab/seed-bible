const callingFrom = that ? that.callingFrom ?? "" : "";
const mainWord = getBot("isMainWord", true);
if(!thisBot.masks.isInsideBible && (thisBot.masks.canBeDraggedBackToParent || callingFrom === "reformSection") && !mainWord.masks.isBibleClosed)
{
    SoundsManager.playIncludeElementInParentSound();
    setTagMask(thisBot, "isBeingIncludedInParent", true);
    const dimension = os.getCurrentDimension();
    let currentBibleStructure = mainWord.masks.bibleStructure.slice();
    if(!thisBot.tags.transformer)
    {
        const currentPositionX = thisBot.masks[dimension + "X"] ?? thisBot.tags[dimension + "X"];
        const currentPositionY = thisBot.masks[dimension + "Y"] ?? thisBot.tags[dimension + "Y"];
        const currentPositionZ = thisBot.masks[dimension + "Z"] ?? thisBot.tags[dimension + "Z"];
        setTag(thisBot, "transformer", getID(mainWord));
        setTagMask(thisBot, dimension + "X", currentPositionX - (mainWord.masks[dimension + "X"] ?? mainWord.tags[dimension + "X"]));
        setTagMask(thisBot, dimension + "Y", currentPositionY - (mainWord.masks[dimension + "Y"] ?? mainWord.tags[dimension + "Y"]));
        setTagMask(thisBot, dimension + "Z", currentPositionZ - (mainWord.masks[dimension + "Z"] ?? mainWord.tags[dimension + "Z"]));
    }
    setTagMask(thisBot, "isInsideBible", true);
    setTagMask(thisBot, "canBeDraggedBackToParent", false);
    setTagMask(thisBot, "isOnTheGround", false);

    for(let testament of currentBibleStructure)
    {
        for(let section of testament.sections)
        {
            if(thisBot.tags.isBookAsSection)
            {
                if(section.botID === getID(thisBot))
                {
                    setTagMask(thisBot, "isInsideTestament", testament.isInsideBible);
                    section.isInsideArrangement = testament.isInsideBible;
                    section.isInsideBible = true;
                    section.isConvertedIntoATray = false;
                }
            }
            else
            {
                for(let piece of section.books)
                {
                    if(Array.isArray(piece))
                    {
                        if(thisBot.tags.isArrangementGroupBook)
                        {
                            for(let book of piece)
                            {
                                if(book.botID === getID(thisBot))
                                {
                                    setTagMask(thisBot, "isInsideTestament", testament.isInsideBible);
                                    setTagMask(thisBot, "isInsideSection", section.isInsideBible);
                                    book.isInsideSection = section.isInsideBible;
                                    book.isInsideArrangement = testament.isInsideBible;
                                    book.isInsideBible = true;
                                    book.isConvertedIntoATray = false;
                                }
                            }
                        }
                    }
                    else
                    {
                        if(piece.botID === getID(thisBot))
                        {
                            setTagMask(thisBot, "isInsideTestament", testament.isInsideBible);
                            setTagMask(thisBot, "isInsideSection", section.isInsideBible);
                            piece.isInsideSection = section.isInsideBible;
                            piece.isInsideArrangement = testament.isInsideBible;
                            piece.isInsideBible = true;
                            piece.isConvertedIntoATray = false;
                        }
                    }
                }
            }
        }
    }

    if(thisBot.tags.isConvertedIntoATray)
    {
        let trayBookOutline = getBot(byTag("isTrayBookOutline", true), byTag("trayBookName", thisBot.tags.bookName));
        if(trayBookOutline)
        {
            ObjectPooler.ReleaseObject({obj: trayBookOutline, tag: trayBookOutline.tags.poolTag});
        }
        setTagMask(thisBot, "isConvertedIntoATray", false);
        // thisBot.lerpColor({startingColor: BibleVizUtils.Functions.HexToRgb(thisBot.masks.color ?? thisBot.tags.color), endingColor: BibleVizUtils.Functions.HexToRgb(thisBot.tags.initialColor), durationInSeconds: 0.75})
        ColorLerper.LerpTag({startingColor: BibleVizUtils.Functions.HexToRgb(thisBot.masks.color ?? thisBot.tags.color), endingColor: BibleVizUtils.Functions.HexToRgb(thisBot.tags.initialColor), durationInSeconds: 0.75, bot: thisBot, tag: BibleVizUtils.Data.tags.InterpolatableColorTags.Color});
    }
    
    setTagMask(mainWord, "bibleStructure", currentBibleStructure);

    if(thisBot.masks.isExpanded)
    {
        thisBot.displayShrinkAnimation();
    }

    if(callingFrom !== "checkProximityToParent")
    {
        setTagMask(thisBot, "isBeingIncludedInParent", false);
    }
     

    return true
}
else
{
    return false;
}
