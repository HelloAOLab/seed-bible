const dimension = os.getCurrentDimension();
const myLobbyId = getBot('lobbyUserBot', true)?.id;
const elementSelections = InstanceManager.GetUsersSelectionForElement({piece: thisBot}).filter((userSelection) => {
    return userSelection.lobbyId != myLobbyId
})

if(elementSelections.length > 0)
{
    if(links.usersNotification)
    {
        setTag(links.usersNotification, "label", elementSelections.length)
    }
    else
    {
        const usersNotification = ObjectPooler.GetObjectFromPool({tag: BibleVizUtils.Data.tags.ObjectPoolTags.UsersNotification});
        const usersNotificationMod = {
            [dimension]: true,
            label: elementSelections.length,
            ownerBotId: thisBot.id
        }
        usersNotification.OnSpawned({mod: usersNotificationMod});
        usersNotification.SetPosition({setX: true, setY: true, setZ: true});
        thisBot.tags.usersNotification = `🔗${usersNotification.id}`
    }
}
else thisBot.TryHideUsersNotification();
