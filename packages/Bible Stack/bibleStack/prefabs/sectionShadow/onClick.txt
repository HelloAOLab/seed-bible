let section;
let mainWord;
if(thisBot.tags.isBookAsSectionShadow)
{
    mainWord = getBot("isMainWord", true);
    section = getBot(byTag("bookName", thisBot.tags.sectionName), byTag("isBook", true));
    section.tryToIncludeInTheBible();
    mainWord.resizeBible();
}
else
{
    section = getBot(byTag("sectionName", thisBot.tags.sectionName), byTag("isSection", true));
    section.reformSection();
}
